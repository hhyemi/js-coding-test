### :heavy_check_mark: 내가 푼 답안

```javascript
function solution(answers) {

    let one = [1, 2, 3, 4, 5];
    let two = [2, 1, 2, 3, 2, 4, 2, 5];
    let three = [3, 3, 1, 1, 2, 2, 4, 4, 5, 5];

    let oneCnt = 0, twoCnt = 0, threeCnt = 0; // 맞은 개수 카운트

    for(let i in answers){

        if(one[i] == null){
           one[i] = one[i-5];
        }
        if(two[i] == null){
           two[i] = two[i-8];
        }        
        if(three[i] == null){
            three[i] = three[i-10];
        }

        if(answers[i] === one[i]){
            oneCnt++;
        }          
        if(answers[i] === two[i]){
            twoCnt++;
        }
        if(answers[i] === three[i]){
            threeCnt++;
        }
    }

    const result2 = [1, 2, 3];
    const result = [oneCnt, twoCnt, threeCnt];

    for(let j=0 ; j< result.length ; j++){
        if(result[j] < result[j+1]){
            result.splice(j,1);
            result2.splice(j,1);
            j--;
        }else if(result[j] > result[j+1]){
            result.splice(j+1,1);
            result2.splice(j+1,1);
            j--;
        }

    }
    return result2;
}
```
배열에 각 답 규칙을 담고 배열이상의 값을 호출하면,, one[6] 같은 5개이상.. 하면 어떻게 할지 고민하다가 null
값이면 각 배열의 lenth를 빼줬다.

### :heavy_check_mark: 참고하면 좋은 다른 답안 

Thanks for https://programmers.co.kr/learn/courses/30/lessons/42840/solution_groups?language=javascript&type=all

```javascript
function solution(answers) {
    var answer = [];
    var a1 = [1, 2, 3, 4, 5];
    var a2 = [2, 1, 2, 3, 2, 4, 2, 5]
    var a3 = [ 3, 3, 1, 1, 2, 2, 4, 4, 5, 5];

    var a1c = answers.filter((a,i)=> a === a1[i%a1.length]).length;
    var a2c = answers.filter((a,i)=> a === a2[i%a2.length]).length;
    var a3c = answers.filter((a,i)=> a === a3[i%a3.length]).length;
    var max = Math.max(a1c,a2c,a3c);

    if (a1c === max) {answer.push(1)};
    if (a2c === max) {answer.push(2)};
    if (a3c === max) {answer.push(3)};

    return answer;
}

```
filter함수로 for문을 대신한 방법이 좋았다.  
```a1[i%a1.length]```.. 나머지연산을 이용해서 하면 쉬운데 내 답안은 뻘짓을 했다.  
마지막에 순서대로  push하는 방법도 참고하면 좋을 것 같다.
